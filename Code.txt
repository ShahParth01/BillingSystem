	}

		return tax;
	}

		return dsct;
	}

	void calculate() {
		float sum;
		x = (get_price() * get_qty());
		z = (x * get_tax()) / 100;
		y = (x + z);
		q = ((y * get_dsct()) / 100);
		sum = (y - q);
		this->sum = sum;
	}

	float get_calculate() {
		return sum;
	}

void gotoxy(int x, int y)
{
	COORD coordinates;
	coordinates.X = x;
	coordinates.Y = y;
	SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), coordinates);
